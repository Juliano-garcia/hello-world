class Block:
    '''A class for block objects'''
    
    def __init__(self, height, length, depth):
        self.height = height
        self.length = length
        self.depth = depth
    
    def get_volume(self):
        '''Method to calculate total volume'''
        return self.height * self.length * self.depth
    
    def cubed(self, volume):
        '''Method to transform the block in a cube based on passed volume'''
        edge = volume**(1/3) # Cube root to get edge size
        self.height = self.length = self.depth = edge
